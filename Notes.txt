public String maskSensitive(Object value, MaskSensitive maskSensitive) {
        if (value instanceof String) {
            return maskSensitiveString((String) value, maskSensitive);
        } else {
            return value.toString();
        }
    }

    private String maskSensitiveString(String value, MaskSensitive maskSensitive) {
        int startPos = maskSensitive.startPos();
        int endPos = maskSensitive.endPos();
        char maskChar = maskSensitive.maskChar();

        StringBuilder maskedValue = new StringBuilder();
        for (int i = 0; i < value.length(); i++) {
            if (i >= startPos && i <= endPos) {
                maskedValue.append(maskChar);
            } else {
                maskedValue.append(value.charAt(i));
            }
        }

        return maskedValue.toString();
    }
