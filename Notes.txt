import { Injectable } from '@angular/core';
import { NativeService } from './native.service';
import { NativeServiceMock } from './native.service.mock';
import { environment } from 'src/environments/environment';

@Injectable({
  providedIn: 'root',
  useFactory: (nativeService: NativeService, nativeServiceMock: NativeServiceMock) => {
    return environment.mockIntegrated ? nativeServiceMock : nativeService;
  },
  deps: [NativeService, NativeServiceMock]
})
export function provideNativeService(
  nativeService: NativeService,
  nativeServiceMock: NativeServiceMock
) {
  return environment.mockIntegrated ? nativeServiceMock : nativeService;
}
